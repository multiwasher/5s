function doPost(e) { try { var data; Logger.log("e.postData.contents: " + e.postData.contents); Logger.log("e.parameter: " + JSON.stringify(e.parameter)); // Tenta ler como formulário (form submission URL-encoded) if (e.parameter && e.parameter.data) { try { var decodedData = decodeURIComponent(e.parameter.data); Logger.log("Dados decodificados: " + decodedData); data = JSON.parse(decodedData); } catch(err) { Logger.log("Erro ao fazer parse de parameter: " + err); } } // Tenta ler como JSON (from fetch) if (!data && e.postData && e.postData.contents && e.postData.contents.length > 0) { try { data = JSON.parse(e.postData.contents); } catch(err) { Logger.log("Erro ao fazer parse de postData: " + err); } } if (!data) { throw new Error("Sem dados recebidos. Parâmetros: " + JSON.stringify(e.parameter)); } Logger.log("Dados finais: " + JSON.stringify(data)); // Se for atualização de Planeamento if (data.action === 'updatePlaneamento') { var planeamentoSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Planeamento'); var range = planeamentoSheet.getDataRange(); var values = range.getValues(); Logger.log("Total de linhas no Planeamento: " + values.length); // Encontra as colunas ID e Status var headers = values[0]; var idColIndex = -1; var statusColIndex = -1; for (var h = 0; h < headers.length; h++) { if (headers[h] === 'ID') idColIndex = h; if (headers[h] === 'Status') statusColIndex = h; } Logger.log("Índices - ID: " + idColIndex + ", Status: " + statusColIndex); Logger.log("Procurando ID: '" + data.id + "'"); // Procura a linha com ID correspondente for (var i = 1; i < values.length; i++) { var sheetId = String(values[i][idColIndex]).trim(); var searchId = String(data.id).trim(); Logger.log("Linha " + i + ": ID='" + sheetId + "'"); if (sheetId === searchId) { Logger.log("ENCONTRADO na linha " + (i + 1) + "! Atualizando STATUS para: '" + data.status + "'"); planeamentoSheet.getRange(i + 1, statusColIndex + 1).setValue(data.status); return HtmlService.createHtmlOutput('OK'); } } Logger.log("ID não encontrado!"); return HtmlService.createHtmlOutput('NOT_FOUND'); } // Se for auditoria 5S (comportamento original) else { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var row = [ new Date(), // A: Data data.posto, // B: Posto data.pontuacao, // C: Pontuação data.estado, // D: Estado data.scores[0], // E: 1.1 data.scores[1], // F: 1.2 data.scores[2], // G: 2.1 data.scores[3], // H: 2.2 data.scores[4], // I: 3.1 data.scores[5], // J: 3.2 data.scores[6], // K: 4.1 data.scores[7], // L: 4.2 data.scores[8], // M: 5.1 data.scores[9], // N: 5.2 JSON.stringify(data) // O: Backup JSON ]; sheet.appendRow(row); return HtmlService.createHtmlOutput('OK'); } } catch (f) { Logger.log("ERRO: " + f.toString()); return HtmlService.createHtmlOutput('ERROR: ' + f.toString()); } }
